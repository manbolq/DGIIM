a0 = 1
a1 = 1

f(x) = a0 * x + a1
fit f(x) 'datos.dat' u 1:2 via a0, a1
stats "datos.dat" u 2:(f($1)) name "F"
varf = F_correlation * F_correlation
title_f = sprintf('Alg. basico R^2=%4.4f',varf)
set term png size 1280,720
set output "comp_k_const"

a4 = 1
a5 = 1


g(x) = a4 * x + a5
fit g(x) 'datos.dat' u 1:2 via a4, a5
stats "datos.dat" u 3:(g($1)) name "G"
varg = G_correlation * G_correlation
title_g = sprintf('Alg. DyV R^2=%4.4f',varg)
set term png size 1280,720
set output "comp_k_const"


set title "Comparacion: k constante"
set xlabel "Numero de componentes"
set ylabel "Tiempo (seg)"
set style line 1 linewidth 2 linecolor rgb "red"
set style line 2 linewidth 2 linecolor rgb "green"
set style line 3 linewidth 2 linecolor rgb "blue"
set style line 4 linewidth 2 linecolor rgb "yellow"



plot f(x) with lines linestyle 1 t title_f, 'datos.dat' u 1:2 w p pt 7 ps 1.5 lc rgb "red", g(x) with lines linestyle 2 t title_g, 'datos.dat' u 1:3 w p pt 7 ps 1.5 lc rgb "green"
